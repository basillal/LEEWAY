    filteredEmployees: typeof this.employees = [];
    displayedEmployees: typeof this.employees = [];
  
    // Department filter
    departmentSearchTerm: string = '';
    selectedDepartments: { [key: string]: boolean } = {};
    allDepartments: string[] = [];
  
    // Attendance filter
    currentAttendanceFilter: string[] = [];
    attendanceFilters = [
      { label: 'Absent', value: ['ABSENT', 'AB PM', 'AB AM'], checked: false },
      { label: 'Casual Leaves', value: ['CLFD', 'CLHD AM', 'CLHD PM'], checked: false },
      { label: 'On Duty Leaves', value: ['OODFD', 'OODHD AM', 'OODHD PM'], checked: false },
      { label: 'Late In', value: 'LATE_PUNCH_IN', checked: false },
      { label: 'Early Out', value: 'EARLY_PUNCH_OUT', checked: false },
      { label: 'Others', value: 'OTHERS', checked: false },
    ];
  
    // Active status filter
    activeFilters = [
      { value: 'ALL', checked: false },
      { label: 'Active Employees', value: true, checked: false },
      { label: 'Inactive Employees', value: false, checked: false },
    ];
  
    // Search and pagination
    searchTerm: string = '';
    employeeCount = this.employees.length;
    currentPage = 1;
    pageSize = 8; // Show 8 employees per page
    totalPages = Math.ceil(this.employeeCount / this.pageSize);
  isAllSelected: boolean = false;
  
    constructor(private readonly router: Router, private readonly cdr: ChangeDetectorRef) {
      this.updateDisplayedEmployees();
      this.initializeDepartments();
    }
  
    ngOnInit(): void {
      this.filteredEmployees = this.employees;
      this.employeeCount = this.employees.length;
      this.totalPages = Math.ceil(this.employees.length / this.pageSize);
      this.initializeDepartments();
      this.updateDisplayedEmployees();
    }
  
    // Initialize departments from employee data
    initializeDepartments(): void {
      this.allDepartments = [...new Set(this.employees.map((emp) => emp.department))].sort((a, b) =>
        a.localeCompare(b)
      );
      this.allDepartments.forEach((dept) => {
        if (!this.selectedDepartments.hasOwnProperty(dept)) {
          this.selectedDepartments[dept] = false;
        }
      });
    }
  
    // Update displayed employees based on filters and pagination
    updateDisplayedEmployees(): void {
      let filtered = this.employees;
  
      // Apply active status filter
      if (this.currentActiveFilter !== null) {
        filtered = filtered.filter((employee) => employee.activeStatus === this.currentActiveFilter);
      }
  
      // Apply attendance filter
      if (this.currentAttendanceFilter.length > 0) {
        filtered = filtered.filter((employee) => {
          const statusList = employee.todayAttendance.statusList || [];
          return this.currentAttendanceFilter.some((status) => statusList.includes(status));
        });
      }
  
      // Apply department filter
      const selectedDepartments = Object.keys(this.selectedDepartments).filter(
        (key) => this.selectedDepartments[key]
      );
      if (selectedDepartments.length > 0) {
        filtered = filtered.filter((employee) => selectedDepartments.includes(employee.department));
      }
  
      // Apply search term filter
      if (this.searchTerm) {
        filtered = filtered.filter(
          (employee) =>
            employee.name.toLowerCase().includes(this.searchTerm.toLowerCase()) ||
            employee.email.toLowerCase().includes(this.searchTerm.toLowerCase()) ||
            employee.empId.toLowerCase().includes(this.searchTerm.toLowerCase())
        );
      }
  
      // Update filteredEmployees and pagination
      this.filteredEmployees = filtered;
      this.employeeCount = this.filteredEmployees.length;
      this.totalPages = Math.ceil(this.employeeCount / this.pageSize);
  
      // Ensure currentPage is within valid range
      if (this.currentPage > this.totalPages) {
        this.currentPage = 1;
      }
  
      // Slice the filteredEmployees array based on the current page and page size
      const startIndex = (this.currentPage - 1) * this.pageSize;
      const endIndex = startIndex + this.pageSize;
      this.displayedEmployees = this.filteredEmployees.slice(startIndex, endIndex);
    }
  
    // Toggle department dropdown
    toggleDepartmentDropdown(): void {
      this.showDepartmentDropdown = !this.showDepartmentDropdown;
      this.departmentSearchTerm = '';
      this.isAllSelected = false;
    }
  
    // Toggle attendance dropdown
    toggleAttendanceDropdown(): void {
      this.showAttendanceDropdown = !this.showAttendanceDropdown;
    }
  
    // Toggle active employees dropdown
    toggleActiveEmployeesDropdown(): void {
      this.showActiveEmployeesDropdown = !this.showActiveEmployeesDropdown;
    }
  
    // Apply active filter
    applyActiveFilter(filter: any): void {
      if (filter.value === 'ALL') {
        const allChecked = !filter.checked;
        this.activeFilters.forEach((f) => (f.checked = allChecked));
      } else {
        filter.checked = !filter.checked;
      }
      this.updateCurrentActiveFilter();
    }
  
    // Update current active filter
    updateCurrentActiveFilter(): void {
      const selectedStatuses = this.activeFilters
        .filter((filter) => filter.checked && filter.value !== 'ALL')
        .map((filter) => filter.value);
  
      if (selectedStatuses.length === 0) {
        this.currentActiveFilter = null;
      } else if (selectedStatuses.includes(true) && selectedStatuses.includes(false)) {
        this.currentActiveFilter = null;
      } else if (selectedStatuses.includes(true)) {
        this.currentActiveFilter = true;
      } else if (selectedStatuses.includes(false)) {
        this.currentActiveFilter = false;
      }
  
      this.updateDisplayedEmployees();
    }
  
    // Reset all filters
    resetAll(): void {
      this.activeFilters.forEach((filter) => (filter.checked = false));
      this.currentActiveFilter = true;
      this.showActiveEmployeesDropdown = false;
      this.searchTerm = '';
      this.selectedDepartments = {};
      this.allDepartments.forEach((dept) => (this.selectedDepartments[dept] = false));
      this.departmentSearchTerm = '';
      this.isAllSelected = false;
      this.attendanceFilters.forEach((filter) => (filter.checked = false));
      this.currentAttendanceFilter = [];
      this.updateDisplayedEmployees();
      this.cdr.detectChanges();
    }
  
    // Navigate to employee overview
    navigateToOverview(email: string, empid: string): void {
      localStorage.setItem('email', email);
      localStorage.setItem('empid', empid);
      this.router.navigate(['/employee-details/overview']);
    }
  
    // Handle search input
    searchEmployees(event: Event): void {
      const searchTerm = (event.target as HTMLInputElement).value.toLowerCase();
      this.searchTerm = searchTerm;
      this.currentPage = 1;
      this.updateDisplayedEmployees();
    }
  
    // Get pagination range
    getPaginationRange(): (number | string)[] {
      const range: (number | string)[] = [];
      const total = this.totalPages;
      const current = this.currentPage;
      const delta = 2; // Number of pages to show around the current page
  
      // Always show the first page
      range.push(1);
  
      // Add dots if the current page is far from the first page
      if (current - delta > 2) {
        range.push('...');
      }
  
      // Add pages around the current page
      for (
        let i = Math.max(2, current - delta);
        i <= Math.min(total - 1, current + delta);
        i++
      ) {
        range.push(i);
      }
  
      // Add dots if the current page is far from the last page
      if (current + delta < total - 1) {
        range.push('...');
      }
  
      // Always show the last page
      if (total > 1) {
        range.push(total);
      }
  
      console.log("my range:",range);
      
      return range;
    }
  
    // Change page
    changePage(page: number | string): void {
      if (typeof page === 'number') {
        this.currentPage = page;
        this.updateDisplayedEmployees();
      }
    }
  
    // Go to previous page
    previousPage(): void {
      if (this.currentPage > 1) {
        this.currentPage--;
        this.updateDisplayedEmployees();
      }
    }
  
    // Go to next page
    nextPage(): void {
      if (this.currentPage < this.totalPages) {
        this.currentPage++;
        this.updateDisplayedEmployees();
      }
    }
  
    // Get status style
    getStatusStyle(status: string): { bg: string; text: string } {

  filteredEmployees: typeof this.employees = [];
  displayedItems: typeof this.employees = [];

  // Search and pagination
  searchTerm: string = '';
  employeeCount = this.employees.length;
  currentPage = 1;
  pageSize = 8; // Show 8 employees per page
  totalPages = Math.ceil(this.employeeCount / this.pageSize);
  isAllSelected: boolean = false;

  constructor(private readonly router: Router, private readonly cdr: ChangeDetectorRef) {
    this.updateDisplayedEmployees();
  }

  ngOnInit(): void {
    this.filteredEmployees = this.employees;
    this.employeeCount = this.employees.length;
    this.totalPages = Math.ceil(this.employees.length / this.pageSize);
    this.updateDisplayedEmployees();
  }



  // Update displayed employees based on filters and pagination
  updateDisplayedEmployees(): void {
    let filtered = this.employees;


    // Apply search term filter
    if (this.searchTerm) {
      filtered = filtered.filter(
        (employee) =>
          employee.name.toLowerCase().includes(this.searchTerm.toLowerCase()) ||
          employee.email.toLowerCase().includes(this.searchTerm.toLowerCase()) ||
          employee.empId.toLowerCase().includes(this.searchTerm.toLowerCase())
      );
    }

    // Update filteredEmployees and pagination
    this.filteredEmployees = filtered;
    this.employeeCount = this.filteredEmployees.length;
    this.totalPages = Math.ceil(this.employeeCount / this.pageSize);

    // Ensure currentPage is within valid range
    if (this.currentPage > this.totalPages) {
      this.currentPage = 1;
    }

    // Slice the filteredEmployees array based on the current page and page size
    const startIndex = (this.currentPage - 1) * this.pageSize;
    const endIndex = startIndex + this.pageSize;
    this.displayedItems = this.filteredEmployees.slice(startIndex, endIndex);
  }


  // Handle search input
  searchEmployees(event: Event): void {
    const searchTerm = (event.target as HTMLInputElement).value.toLowerCase();
    this.searchTerm = searchTerm;
    this.currentPage = 1;
    this.updateDisplayedEmployees();
  }

  // Get pagination range
  getPaginationRange(): (number | string)[] {
    const range: (number | string)[] = [];
    const total = this.totalPages;
    const current = this.currentPage;
    const delta = 2; // Number of pages to show around the current page

    // Always show the first page
    range.push(1);

    // Add dots if the current page is far from the first page
    if (current - delta > 2) {
      range.push('...');
    }

    // Add pages around the current page
    for (
      let i = Math.max(2, current - delta);
      i <= Math.min(total - 1, current + delta);
      i++
    ) {
      range.push(i);
    }

    // Add dots if the current page is far from the last page
    if (current + delta < total - 1) {
      range.push('...');
    }

    // Always show the last page
    if (total > 1) {
      range.push(total);
    }
    return range;
  }

  // Change page
  changePage(page: number | string): void {
    if (typeof page === 'number') {
      this.currentPage = page;
      this.updateDisplayedEmployees();
    }
  }

  // Go to previous page
  previousPage(): void {
    if (this.currentPage > 1) {
      this.currentPage--;
      this.updateDisplayedEmployees();
    }
  }

  // Go to next page
  nextPage(): void {
    if (this.currentPage < this.totalPages) {
      this.currentPage++;
      this.updateDisplayedEmployees();
    }
  }
